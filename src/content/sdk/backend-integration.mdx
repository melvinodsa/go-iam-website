## Step 3: Implement Authentication

### Verify Authentication Code

<CodeBlock
  copyButton={false}
  codes={[
    {
      language: "go",
      label: "Go",
      code: `ctx := context.Background()
  token, err := service.Verify(ctx, "auth-code") 
  if err != nil {
    log.Fatalf("Failed to verify: %v", err)
  }
  fmt.Println("Access Token:", token)}
`,
    },
      {
      language: "rust",
      label: "Rust",
      code:`use goiam_rust::{new_service, Service, Resource};

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    // Initialize the service
    let service = new_service(
        "https://api.goiam.com".to_string(),
        "your-client-id".to_string(),
        "your-secret".to_string()
    );

    // Verify a code and get access token
    match service.verify("authorization-code").await {
        Ok(token) => {
            println!("Access Token: {}", token);
        },
        Err(e) => eprintln!("Verification failed: {}", e),
    }

    Ok(())
}
`
    },
    {
      language: "python",
      label: "Python",
      code:

    token = service.verify("auth-code")
    print(f"Access Token: {token}")
except Exception as error:
    print(f"Failed to verify code: {error}")
`
  },
    {
language: "typescript",

  ]}
